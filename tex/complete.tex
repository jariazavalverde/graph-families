\documentclass{article}
\usepackage{tikz}

\newcommand{\complete}[1]{
\begin{tikzpicture}[line width=0.4mm]
	\node [circle, minimum size=6cm] (o) {};
	\foreach \x in {1,...,#1}{
		\node[draw, minimum size=5mm, circle] (\x) at (o.\x*360/#1) {};
	}
	\foreach \i in {1,...,#1} {
		\foreach \j in {1,...,#1} {
			\path (\i) edge node {} (\j);
		}
	}
\end{tikzpicture}
}
